28  import java.util.List;
29  import java.util.Map;

31  import javax.servlet.ServletContext;

33  /**
34   * Represents the destination of a <a href="package-summary.html#handlerMethod">handler method</a>.  
35   * This class serves as a unifying facade for redirecting and forwarding in the Servlet API 
36   * and working with request parameters.
37   * 
38   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
39   * 
40   * @see <a href="Destinations.html#forward(java.lang.String)">Destinations.forward()</a>
41   * @see <a href="Destinations.html#redirect(java.lang.String)">Destinations.redirect()</a>
42   */
43  public interface Destination {
44  
45  	/**
46  	 * Adds a parameter to the {@code Destination}.  Multi-valued parameters are supported
47  	 * and will be serialized into a query string with the same parameter name and multiple
48  	 * values, eg: {@code /somePath?dogs=woof&dogs=bark&cats=meow&cats=purr}
49  	 * @param name the name of the parameter
50  	 * @param value the value of the parameter
51  	 */
52  	public abstract void addParameter(String name, String value);
53  
54  	public abstract String getPath();
55  
56  	public abstract void setPath(String path);
57  
58  	public abstract Map<String, List<String>> getParameters();
59  
60  	public abstract void setParameters(Map<String, List<String>> parameters);
61  
62  	public abstract Boolean getRedirect();
63  
64  	public abstract void setRedirect(Boolean redirect);
65  
66  	/**
67  	 * Encodes the {@code Destination} path and available parameters. The parameters have any ampersands
68  	 * replaced with &{@code &amp;amp;} and parameter names and their associated values are sorted (mainly for
69  	 * testing purposes).
70  	 * 
71  	 * @param context the handler context that this destination is running under
72  	 * @return the encoded destination path and query string
73  	 */
74  	public abstract String encode(ServletContext context);
75  
76  }
