26  package agave;

28  import java.io.File;
29  
30  /**
31   * A part in a multipart/form-data HTTP post which either encapsulates a {@code File} upload or a 
32   * {@code String} parameter.  In the case of a {@code String} parameter this can have the name and 
33   * the parameter value set.  In the case of a {@code File} part, the name, filename, content type,
34   * and contents will be set.
35   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
36   */
37  public interface Part {
38  
39      public String getName();
40      public void setName(String name);
41      public String getContentType();
42      public void setContentType(String contentType);
43      public String getFilename();
44      public void setFilename(String filename);
45      public String getParameterValue();
46      public void setParameterValue(String parameterValue);
47      public void addHeader(String headerName, String headerValue);
48      public String getHeader(String headerName);
49      public File getContents();
50      public void setContents(File contents);
51  
52  }
