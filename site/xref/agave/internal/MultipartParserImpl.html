48  public class MultipartParserImpl implements MultipartParser {

50      private static class CoupledLine {
51          private StringBuilder characters = new StringBuilder();
52          private List<Byte> bytes = new LinkedList<Byte>();
53  
54          public void append(int i) {
55              characters.append((char)i);
56              bytes.add((byte)i);
57          }
58      }

60      private static final String DEFAULT_SUFFIX = ".tmp";
61      private static final Pattern BOUNDARY_PATTERN = Pattern.compile("multipart/form-data;\\s*boundary=(.*)");
62      private static final String CONTENT_DISPOSIITON = "Content-Disposition:\\s*form-data;\\s*name=\"(.*)\"";
63      private static final Pattern PART_PATTERN = Pattern.compile(CONTENT_DISPOSIITON + ";\\s*filename=\"(.+)\"");
64      private static final Pattern FILENAME_PATTERN = Pattern.compile("(.*)\\.(.*)");
65      private static final Pattern PARAMETER_PATTERN = Pattern.compile(CONTENT_DISPOSIITON);
66      private static final Pattern CONTENT_TYPE_PATTERN = Pattern.compile("Content-Type:\\s*(.+)");
67      private static final Pattern OTHER_HEADER_PATTERN = Pattern.compile("(\\S+):\\s*(.+?)");
68      
69      private Map<String, Collection<String>> parameters;
70      private Map<String, Part> parts;
71      private String boundary;
72      private String eos;
73      private InputStream in;

75      public MultipartParserImpl(HttpServletRequest request) throws IOException {
76          parameters = new HashMap<String, Collection<String>>();
77          parts = new HashMap<String, Part>();
78          Matcher boundaryMatcher = BOUNDARY_PATTERN.matcher(request.getContentType());
79          if (boundaryMatcher.matches() && boundaryMatcher.groupCount() >= 1) {
80              boundary = "--" + boundaryMatcher.group(1);
81          }
82          eos = boundary + "--";
83          in = request.getInputStream();
84          readLine(in);
85      }

99      public void parseInput() throws IOException {
100         try {
101             while (true) {
102                 Part part = new PartImpl();
103                 readHeaders(part);
104                 if (part.getFilename() != null) {
105                     if (readPart(part)) {
106                         break;
107                     }
108                 } else {
109                     if (readParameter(part)) {
110                         break;
111                     }
112                 }
113             }
114         } finally {
115             in.close();
116             in = null;
117         }
118     }

120     public void readHeaders(Part part) throws IOException {
121         String line = null;
122         while ((line = readLine(in)) != null) {
123             line = line.trim();
124             
125             if ("".equals(line)) {
126                 break;
127             }
128             
129             Matcher matcher = PART_PATTERN.matcher(line);
130             if (matcher.matches() && matcher.groupCount() >= 2) {
131                 part.setName(matcher.group(1));
132                 part.setFilename(matcher.group(2));
133                 continue;
134             }
135             
136             matcher = PARAMETER_PATTERN.matcher(line);
137             if (matcher.matches() && matcher.groupCount() >= 1) {
138                 part.setName(matcher.group(1));
139                 continue;
140             }
141             
142             matcher = CONTENT_TYPE_PATTERN.matcher(line);
143             if (matcher.matches() && matcher.groupCount() >= 1) {
144                 part.setContentType(matcher.group(1));
145                 continue;
146             }
147             
148             matcher = OTHER_HEADER_PATTERN.matcher(line);
149             if (matcher.matches() && matcher.groupCount() >= 2) {
150                 part.addHeader(matcher.group(1), matcher.group(2));
151                 continue;
152             }
153         }
154     }

156     private String readLine(InputStream in) throws IOException {
157         StringBuilder text = new StringBuilder();
158         
159         int i = -1;
160         while ((i = in.read()) != -1) {
161             text.append((char) i);
162             
163             if ((char) i == '\n') {
164                 break;
165             }
166         }
167         
168         return text.toString();
169     }

171     private CoupledLine readCoupledLine(InputStream in) throws IOException {
172         CoupledLine line = new CoupledLine();
173         
174         int i = -1;
175         while ((i = in.read()) != -1) { 
176             line.append(i);
177             
178             if ((char) i == '\n') {
179                 break;
180             }
181         }
182         
183         return line;
184     }

186     public boolean readParameter(Part part) throws IOException {
187         boolean end = false;
188 
189         StringBuilder parameterValue = new StringBuilder();
190         String line = null;
191         
192         while ((line = readLine(in)) != null) {
193             line = line.trim();
194             
195             if (eos.equals(line)) {
196                 end = true;
197                 break;
198             }
199             
200             if (boundary.equals(line)) {
201                 break;
202             }
203             
204             parameterValue.append(line);
205         }
206         
207         if (!parameters.containsKey(part.getName())) {
208             parameters.put(part.getName(), new ArrayList<String>());
209         }
210         parameters.get(part.getName()).add(parameterValue.toString());
211         
212         return end;
213     }
214     
215     public boolean readPart(Part part) throws IOException {
216         boolean end = false;
217 
218         String prefix = null;
219         String suffix = null;
220         
221         Matcher matcher = FILENAME_PATTERN.matcher(part.getFilename());
222         if (matcher.matches() && matcher.groupCount() >= 2) {
223             prefix = matcher.group(1);
224             suffix = (matcher.group(2).startsWith(".")) ? matcher.group(2) : '.' + matcher.group(2);
225         } else {
226             prefix = part.getName();
227             suffix = DEFAULT_SUFFIX;
228         }
229         
230         File temporaryFile = File.createTempFile(prefix, suffix);
231         temporaryFile.deleteOnExit();
232         
233         FileOutputStream out = new FileOutputStream(temporaryFile);
234         
235         CoupledLine line = null;
236         while ((line = readCoupledLine(in)) != null) {
237             String text = line.characters.toString().trim();
238             
239             if (eos.equals(text)) {
240                 end = true;
241                 break;
242             }
243             
244             if (boundary.equals(text)) {
245                 break;
246             }
247             
248             for (Byte b : line.bytes) {
249                 out.write(b);
250             }
251         }
252         
253     	out.flush();
254         out.close();
255         part.setContents(temporaryFile);
256         parts.put(part.getName(), part);
257         return end;
258     }

260     protected void finalize() throws Exception {
261         if (in != null) {
262             in.close();
263         }
264     }
