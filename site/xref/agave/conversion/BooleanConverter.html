26  package agave.conversion;
27  
28  import agave.exception.ConversionException;
29  import java.util.Locale;
30  
31  /**
32   * Converts an input {@code String} into a {@code Boolean} object.
33   * @author <a href="mailto:damiancarrillo@gmail.com">Damian Carrillo</a>
34   */
35  public class BooleanConverter implements StringConverter<Boolean> {
36  
37      /**
38       * Performs the conversion.
39       * @param input the input parameter as a {@code String}. Supported inputs are (none are case sensitive): 
40       *  <p>
41       *    <strong>Input values that resolve to {@code Boolean.TRUE}:</strong>
42       *    <ul>
43       *      <li>true</li>
44       *      <li>t</li>
45       *      <li>1</li>
46       *      <li>yes</li>
47       *      <li>y</li>
48       *    </ul>
49       *  </p>
50       *  <p>
51       *    <strong>Input values that resolve to {@code Boolean.FALSE}:</strong>
52       *    <ul>
53       *      <li>false</li>
54       *      <li>f</li>
55       *      <li>0</li>
56       *      <li>no</li>
57       *      <li>n</li>
58       *    </ul>
59       *  </p>
60       * @return a {@code Boolean} object representing the truth value of the input
61       * @throws ConversionException when an unsupported input string is supplied as an argument
62       */ 
63      public Boolean convert(String input, Locale locale) throws ConversionException {
64          Boolean value = null;
65          if (input != null && !"".equals(input)) {
66              String parameter = input.toLowerCase();
67              if ("true".equals(parameter)
68                  || "t".equals(parameter)
69                  || "1".equals(parameter)
70                  || "on".equals(parameter)
71                  || "yes".equals(parameter)
72                  || "y".equals(parameter)) {
73                  value = Boolean.TRUE;
74              } else if ("false".equals(parameter)
75                  || "f".equals(parameter)
76                  || "0".equals(parameter)
77                  || "off".equals(parameter)
78                  || "no".equals(parameter)
79                  || "n".endsWith(parameter)) {
80                  value = Boolean.FALSE;
81              } else {
82                  throw new ConversionException("Could not convert " + input + " into a boolean object");
83              }
84          }
85          return value;
86      }

88  }
